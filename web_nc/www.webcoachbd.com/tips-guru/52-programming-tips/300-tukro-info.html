<?php
$x = 9;
$y = 5;
return;
//followings will not be executed
if($x > $y){
    echo 'stuff';
}
?>



তথ্য ২
SQL কোয়েরিতে কখনও প্রশ্নবোধক চিহ্ন (?) দেখেছেন? এটাকে বলে placeholder.
সাধারনত PDO তে কোয়েরি লেখার সময় এই চিহ্ন দিয়ে প্যারামিটারের কাজ করা হয়। অর্থ্যাৎ কোয়েরিতে এটা থাকা মানে সেখানে যেকোন মান পাঠাতে পারবেন। যেমন

<?php
$stm = $dbh->prepare('SELECT *FROM users WHERE salary < ? AND age = ?');
$stm->execute(array('45k', '30'));

//now execute and fetch result
?>



তথ্য ৩

 

জেকোয়েরি দিয়ে data এট্রিবিউটের মান ধরার সময় $('#user_id').data('agent_id') এভাবে অবশ্যই ধরবেন না যখন data এট্রিবিউটের মান আপডেট হবে on the fly. বরং ধরতে হবে এভাবে $('#user_id').attr('data-agent_id');

var user_id = parseInt($('#user_id').attr('data-agent_id'));



তথ্য ৪

জেকোয়েরিতে যদি কখনও integer কে string বানানোর দরকার হয় তাহলে integer সামনে শুধু একটা কোটেশন দিয়ে দিন। যেমন var x = 9; x = ''+9;
প্রথমে x পূর্নসংখ্যা ছিল কিন্তু ২য় বার স্ট্রিং হয়ে গেছে। বিশ্বাস না হয় টেস্ট করে দেখুন ফায়ারবাগে। x এর মান এখন '9' হয়ে গেছে। যেমন


var x = 9;
x = ''+9;
//now x is '9'


তথ্য ৫
যখন ডাইনামিক কনটেন্ট এ কোন ইভেন্ট ঘটলে কিছু করতে চান তখন সাধারনত on দিয়ে কাজ করেন তবে মাঝে মাঝে এটা কাজ করবেনা তখন live দিয়ে কাজ করতে হবে। যেমন 

$('#dynamic_content').live('click',function(){
    //here code
});


তবে আগে থেকে migrate লাইব্রেরিটি (ডাউনলোড লিংক) যোগ করে নিতে হবে live ব্যবহার করলে।


শেষ তথ্য
কোডইগনাইটারে কখনই config.php তে base_url এ হাত দিবেন না। যখন রিসোর্সের (যেমন ইমেজ) লিংক দিবেন তখন base_url('assets/css/style.css') আর যখন সাইটের লিংক দিতে হবে তখন site_url('controller/method') এভাবে দিলেই কোডইগনাইটার অটোমেটিক আসল লিংক বের করে নিবে, লোকাল কিংবা প্রডাকশন সার্ভারে।
